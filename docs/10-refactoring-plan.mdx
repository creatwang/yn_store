### 1.1 每个模块需要实现的文件清单

```
模块 X（例如 product）：

✅ packages/db/src/schema/{module}.ts       → Drizzle 表定义（如已存在，补充缺失表）
✅ packages/validators/src/{module}.ts      → Zod schema（create/update/list query）
✅ apps/server/src/routes/admin/{module}.ts → Hono 链式路由（路径对齐 Medusa）
✅ apps/server/src/services/{module}.service.ts → 业务逻辑（本模块查询，不跨模块 JOIN）
✅ apps/admin/src/hooks/api/{module}.tsx    → 从 Dashboard 拷贝 hooks
✅ apps/admin/src/lib/client.ts             → SDK 适配方法（list/retrieve/create/update/delete）
✅ apps/admin/src/routes/{module}/           → 从 Dashboard 拷贝 UI 页面
```

### 1.2 命名规范

| 元素 | 规范 | 示例 |
|------|------|------|
| 路由文件 | 驼峰英文，复数 | `products.ts`, `orders.ts` |
| Service 文件 | `{entity}.service.ts` | `product.service.ts` |
| Schema 文件 | 按业务域命名 | `product.ts`, `inventory.ts` |
| Validator 文件 | 按业务域命名 | `product.ts` |

### 1.3 代码模板

#### 路由文件模板 (`routes/admin/{module}.ts`)

```typescript
import { Hono } from "hono"
import { zValidator } from "@hono/zod-validator"
import { {Module}Service } from "../../services/{module}.service"
import { adminAuth, type AuthVariables } from "../../middleware/auth"

export const admin{Modules} = new Hono<{ Variables: AuthVariables }>()
  .use("*", adminAuth)
  .get("/", async (c) => {
    const result = await {module}Service.list(/* query params */)
    return c.json(result)
  })
  .get("/:id", async (c) => {
    const result = await {module}Service.getById(c.req.param("id"))
    return c.json(result)
  })
  .post("/", async (c) => {
    const body = await c.req.json()
    const result = await {module}Service.create(body)
    return c.json(result, 201)
  })
  .post("/:id", async (c) => {
    const body = await c.req.json()
    const result = await {module}Service.update(c.req.param("id"), body)
    return c.json(result)
  })
  .delete("/:id", async (c) => {
    await {module}Service.delete(c.req.param("id"))
    return c.json({ id: c.req.param("id"), object: "{entity}", deleted: true })
  })
```

#### Service 文件模板 (`services/{module}.service.ts`)

```typescript
import { and, count, eq, isNull } from "drizzle-orm"
import { getDb, {table} } from "@my-store/db"
import { HTTPException } from "hono/http-exception"

export const {module}Service = {
  async list(query: { limit?: number; offset?: number }) {
    const db = getDb()
    const conditions = [isNull({table}.deleted_at)]
    const where = and(...conditions)
    const [rows, [{ total }]] = await Promise.all([
      db.select().from({table}).where(where).limit(query.limit ?? 50).offset(query.offset ?? 0),
      db.select({ total: count() }).from({table}).where(where),
    ])
    return { {entities}: rows, count: Number(total), limit: query.limit ?? 50, offset: query.offset ?? 0 }
  },

  async getById(id: string) {
    const db = getDb()
    const [item] = await db.select().from({table}).where(eq({table}.id, id)).limit(1)
    if (!item) throw new HTTPException(404, { message: "{Entity} not found" })
    return { {entity}: item }
  },

  async create(input: any) {
    const db = getDb()
    const id = generateId("{prefix}")
    const [created] = await db.insert({table}).values({ id, ...input }).returning()
    return { {entity}: created }
  },

  async update(id: string, input: any) {
    const db = getDb()
    const [updated] = await db.update({table})
      .set({ ...input, updated_at: sql`now()` })
      .where(eq({table}.id, id))
      .returning()
    if (!updated) throw new HTTPException(404, { message: "{Entity} not found" })
    return { {entity}: updated }
  },

  async delete(id: string) {
    const db = getDb()
    await db.update({table})
      .set({ deleted_at: sql`now()` })
      .where(eq({table}.id, id))
    return { deleted: true }
  },
}
```
