# 自动化测试方案 — 跨境电商平台

> 分后端 API 测试、前端 E2E 测试、DB 层测试三个维度。按优先级逐步实施。

---

## 1. 后端 API 集成测试（优先实施）

### 技术选型

| 工具 | 用途 |
|------|------|
| **Vitest** | 测试运行器（与 Vite 生态一致，速度快） |
| **Hono Test Client** | `testClient(app, baseUrl)` — Hono 内置的 HTTP 测试工具 |
| **@my-store/db** | 直连测试数据库 |

### 安装

```bash
pnpm add -D vitest @hono/node-server -w --filter=@my-store/server
```

### 测试数据库

```bash
# .env.test
DATABASE_URL=postgres://postgres:postgres@localhost:5432/medusa_test
JWT_SECRET=test-secret-key
```

建议用 Docker 快速起一个干净的测试库：
```bash
docker run -d --name medusa-test-db \
  -e POSTGRES_PASSWORD=postgres \
  -p 5433:5432 postgres:16
```


```typescript
  const res = await client.api.auth.user.emailpass.$post({

afterAll(async () => {
```

### 产品模块测试示例

```typescript
// tests/admin/products.test.ts
import { describe, it, expect, beforeEach } from "vitest"
import { client, getAuthToken, seedProduct } from "../setup"

describe("Admin Products API", () => {
  let token: string

  beforeEach(async () => {
    token = await getAuthToken()
  })

  describe("GET /api/admin/products", () => {
    it("返回产品列表和分页信息", async () => {
      await seedProduct({ title: "Test Product" })
      const res = await client.api.admin.products.$get(
        { query: { limit: "10", offset: "0" } },
        { headers: { Authorization: `Bearer ${token}` } }
      )
      expect(res.status).toBe(200)
      const body = await res.json()
      expect(body.products).toBeInstanceOf(Array)
      expect(body.count).toBeGreaterThanOrEqual(0)
    })

    it("q 参数模糊搜索", async () => {
      await seedProduct({ title: "UniqueName123" })
      const res = await client.api.admin.products.$get({
        query: { q: "UniqueName123" }
      })
      expect(res.status).toBe(200)
      const body = await res.json()
      expect(body.products.length).toBeGreaterThanOrEqual(1)
    })

    it("未登录返回 401", async () => {
      const res = await client.api.admin.products.$get()
      expect(res.status).toBe(401)
    })
  })

  describe("GET /api/admin/products/:id", () => {
    it("返回产品详情含关联数据", async () => {
      const { product } = await seedProduct({ title: "Detail Test" })
      const res = await client.api.admin.products[":id"].$get({
        param: { id: product.id }
      })
      expect(res.status).toBe(200)
      const body = await res.json()
      expect(body.product.title).toBe("Detail Test")
      expect(body.product.variants).toBeInstanceOf(Array)
      expect(body.product.options).toBeInstanceOf(Array)
    })

    it("不存在返回 404", async () => {
      const res = await client.api.admin.products[":id"].$get({
        param: { id: "nonexistent" }
      })
      expect(res.status).toBe(404)
    })
  })

  describe("POST /api/admin/products", () => {
    it("创建产品成功返回 201", async () => {
      const res = await client.api.admin.products.$post({
        json: { title: "New Product" }
      })
      expect(res.status).toBe(201)
      const body = await res.json()
      expect(body.product.id).toBeTruthy()
      expect(body.product.title).toBe("New Product")
    })
  })

  describe("POST /api/admin/products/:id", () => {
    it("更新产品成功", async () => {
      const { product } = await seedProduct({ title: "Old Title" })
      const res = await client.api.admin.products[":id"].$post({
        param: { id: product.id },
        json: { title: "New Title" }
      })
      expect(res.status).toBe(200)
      const body = await res.json()
      expect(body.product.title).toBe("New Title")
    })
  })

  describe("DELETE /api/admin/products/:id", () => {
    it("软删产品成功", async () => {
      const { product } = await seedProduct({ title: "To Delete" })
      const res = await client.api.admin.products[":id"].$delete({
        param: { id: product.id }
      })
      expect(res.status).toBe(200)
      const body = await res.json()
      expect(body.deleted).toBe(true)
    })
  })
})
```

### 变体模块测试示例

```typescript
// tests/admin/variants.test.ts
describe("Admin Product Variants API", () => {
  it("创建变体", async () => {
    const { product } = await seedProduct()
    const res = await client.api.admin.products[":productId"].variants.$post({
      param: { productId: product.id },
      json: { title: "Red", sku: "RED-001" }
    })
    expect(res.status).toBe(201)
  })

  it("变体列表不包含 inventory_items", async () => {
    const { product } = await seedProduct()
    const res = await client.api.admin.products[":productId"].variants.$get({
      param: { productId: product.id }
    })
    const body = await res.json()
    for (const variant of body.variants) {
      // 对齐 Medusa 官方 defaultAdminProductVariantFields
      expect(variant.inventory_items).toBeUndefined()
    }
  })
})
```

## 2. 前端 E2E 测试

### 技术选型

| 工具 | 用途 |
|------|------|
| **Playwright** | 浏览器自动化 |
| **@playwright/test** | 测试框架 |

### 安装

```bash
pnpm add -D @playwright/test -w
npx playwright install chromium
```

### 目录

```
tests/
  e2e/
    login.spec.ts
    products.spec.ts
    orders.spec.ts
    fixtures/
      seed.ts        # 测试数据准备
```

### 示例

```typescript
// tests/e2e/products.spec.ts
import { test, expect } from "@playwright/test"

test.describe("产品管理", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5173/app/login")
    await page.fill('input[name="email"]', "admin@medusa-test.com")
    await page.fill('input[name="password"]', "supersecret")
    await page.click('button[type="submit"]')
    await page.waitForURL("**/app/**")
  })

  test("产品列表页加载", async ({ page }) => {
    await page.goto("http://localhost:5173/app/products")
    await expect(page.getByText("产品")).toBeVisible()
    // 验证表格有数据行
    const rows = page.locator("table tbody tr")
    await expect(rows.first()).toBeVisible({ timeout: 5000 })
  })

  test("创建产品", async ({ page }) => {
    await page.goto("http://localhost:5173/app/products/create")
    await page.fill('input[name="title"]', "E2E Test Product")
    await page.click('button[type="submit"]')
    await page.waitForURL("**/app/products/prod_*")
    await expect(page.getByText("E2E Test Product")).toBeVisible()
  })

  test("产品详情页查看", async ({ page }) => {
    await page.goto("http://localhost:5173/app/products/prod_01J...")
    await expect(page.getByText("常规信息")).toBeVisible()
    await expect(page.getByText("变体")).toBeVisible()
    await expect(page.getByText("选项")).toBeVisible()
    await expect(page.getByText("媒体")).toBeVisible()
  })
})
```

## 3. 测试优先级

| 阶段 | 测试类型 | 范围 | 投入 |
|:----:|---------|------|:----:|
| **立即** | 后端集成测试 | products, auth | 2h |
| P0 | 后端集成测试 | orders, customers | 4h |
| P1 | 后端集成测试 | inventory, regions, sales-channels | 4h |
| P1 | E2E 冒烟测试 | 登录 + 产品列表 + 订单列表 | 3h |
| P2 | 后端集成测试 | pricing, promotions, shipping | 8h |
| P2 | E2E 全流程 | 创建产品→下单→支付 | 8h |
